Пример 2: Контроль джостика кнопками

Инициализируем переменную окружения.

Создаем джостик.

Отрисовываем джостик.

Отлавливаем нажатие стрелочки влево или вправа.

Изменяем положение джостика.

 
    var pjs = new PointJS('2D', 800, 600, { backgroundColor: '#DDDDDD' });
    var djostik = pjs.game.newImageObject({
        file : 'djostik.png',
        h : 100 * r, 
        onload : function () {
            this.y = -this.h + height + 20*r;
        }
    });
    djostik.draw();
